int count = 0;
while (count++ < 50)
beans.add(new TestBean(count,”我的是ID为”+ count +”的成员”));

beans.stream()
.filter(testBean ->(testBean.getId()>20))
.forEach(testBean -> {
System.out.println(testBean);
});

beans.stream()
.filter(testBean ->(testBean.getId()>20))
.map(TestBean::getContent)
.forEach(testBean -> {
